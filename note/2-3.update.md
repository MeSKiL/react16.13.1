/**
 *
 * @param expirationTime
 * @param suspenseConfig
 * @description 创建update并返回
 * @returns {Update<*>}
 */
export function createUpdate(
  expirationTime: ExpirationTime,
  suspenseConfig: null | SuspenseConfig,
): Update<*> {
  const update: Update<*> = {
    expirationTime,
    suspenseConfig,

    tag: UpdateState,
    // UpdateState = 0 更新state
    // ReplaceState = 1 替代state
    // ForceState = 2 强制更新
    // CaptureUpdate = 3 捕获错误update

    payload: null,
    // payload:实际操作的内容,除此渲染是{element},setState的话是第一个参数
    callback: null,
    next: null,
    // next:下一个更新
  };
  if (__DEV__) {
    update.priority = getCurrentPriorityLevel();
  }
  return update;
}
/**
 *
 * @param fiber rootFiber
 * @param update createUpdate出的update对象
 * @description 往updateQueue环中插入新update的过程
 */
export function enqueueUpdate<State>(fiber: Fiber, update: Update<State>) {
  // createFiberRoot的时候init的updateQueue并且pending是null
  const updateQueue = fiber.updateQueue;
  if (updateQueue === null) {
    // Only occurs if the fiber has been unmounted.
    return;
  }

  const sharedQueue: SharedQueue<State> = (updateQueue: any).shared;
  // 此次update到来前 最新的update，指向最旧的update (环尾的update)
  const pending = sharedQueue.pending;
  
  // first.next = first
  //
  // second.next = first
  // first.next = second
  //
  // third.next = first
  // second.next = third
  //
  // forth.next = first
  // third.next = forth
  
  // fifth.next = first
  // forth.next = fifth
  
  if (pending === null) {
    // 第一次更新update自己指向自己就行了
    // This is the first update. Create a circular list.
    update.next = update;
  } else {
    // 新update指向环头的update
    update.next = pending.next;
    // 环尾的update指向新update
    pending.next = update;
    // 最新的update插入到环尾并且指向环头
  }
  sharedQueue.pending = update;

  if (__DEV__) {
    // xxx
  }
}

- 多个update可以同时存在
